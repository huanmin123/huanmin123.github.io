SSM项目基于AOP日志管理

准备工作

使用AOP日志管理需要在MVC中添加日志的AOP注解支持

最好就是spring和MVC都添加一个AOP 注解支持 这样防止有些使用AOP功能的 没有效果

<aop:aspectj-autoproxy proxy-target-class="true"/>

至于导入的Maven就不用多说了 因为到这一步 基本你的项目已经完成了 应该jar导入的差不多了

使用AOP必须导入

    <dependency>
        <groupId>org.aspectj</groupId>
        <artifactId>aspectjweaver</artifactId>
        <version>1.6.8</version>
    </dependency>

ISysLogDao

public interface ISysLogDao {
    @InsertProvider(type = SQL_LogDao.class ,method = "save")
    public void save(SysLog sysLog) throws Exception;
    @Select("select * from sysLog")
    List<SysLog> findAll() throws Exception;

}

SQL_LogDao

public class SQL_LogDao {
    public String save() {
        SQL sql = new SQL();
        String uuid = UUID.randomUUID().toString().replaceAll("-", "");
        sql.INSERT_INTO("syslog").INTO_COLUMNS("id", "visitTime", "username", "ip", "url", "executionTime", "method", "methodParType", "methodParName", "methodParvalue", "methodReturn")
                .INTO_VALUES("'" + uuid + "'", "#{visitTime}", "#{username}", "#{ip}", "#{url}", "#{executionTime}", "#{method}", "#{methodParType}", "#{methodParName}", "#{methodParvalue}", "#{methodReturn}");
        return sql.toString();
    }
}

utils层

然后就是 AOP了 我将这个文件放在utils工具层了

LogAop

package cn.xh.utils;

import cn.xh.entity.SysLog;
import cn.xh.service.ISysLogService;
import org.aspectj.lang.JoinPoint;
import org.aspectj.lang.annotation.After;
import org.aspectj.lang.annotation.Aspect;
import org.aspectj.lang.annotation.Before;
import org.springframework.beans.factory.annotation.Autowired;
import org.springframework.core.LocalVariableTableParameterNameDiscoverer;
import org.springframework.security.core.context.SecurityContext;
import org.springframework.security.core.context.SecurityContextHolder;
import org.springframework.security.core.userdetails.User;
import org.springframework.stereotype.Component;

import javax.servlet.http.HttpServletRequest;
import java.lang.reflect.Method;
import java.lang.reflect.Type;
import java.util.Arrays;
import java.util.Date;

@Component
@Aspect
public class LogAop {

    @Autowired
    private HttpServletRequest request;

    @Autowired
    private ISysLogService sysLogService;

    private Date visitTime; //开始时间
    private Class clazz; //访问的类
    private String method;//访问的方法全路径
    private String methodParType;//获取方法参数类型
    private String methodParName;//获取方法参数名
    private String methodParvalue;//获取方法参数值
    private String methodReturn;//获取方法返回值


    //前置通知  主要是获取开始时间，执行的类是哪一个，执行的是哪一个方法
    @Before("execution(* cn.xh.web.*.*(..))")
    public void doBefore(JoinPoint jp) throws ClassNotFoundException, NoSuchMethodException {
        visitTime = new Date();//当前时间就是开始访问的时间
        clazz = jp.getTarget().getClass(); //具体要访问的类
        String methodName = jp.getSignature().getName(); //获取访问的方法的名称
        //方法全路径
        method = jp.getTarget().getClass().getName() + "." + methodName;

        Object[] args = jp.getArgs();//获取访问的方法的参数 值
        log4jutils.logger.info("参数值:" + Arrays.toString(args));
        methodParvalue = Arrays.toString(args);
        Class[] paramTypes = null;  //获取参数类型
        Method[] methods = clazz.getMethods();
        for (int i = 0; i < methods.length; i++) {
            if (methodName.equals(methods[i].getName())) {//和传入方法名匹配
                Class[] params = methods[i].getParameterTypes();
                paramTypes = new Class[params.length];
                for (int j = 0; j < params.length; j++) {
                    paramTypes[j] = Class.forName(params[j].getName());
                }
                break;
            }
        }
        log4jutils.logger.info("参数的类型:" + Arrays.toString(paramTypes));
        methodParType = Arrays.toString(paramTypes);

        //spring中的 类  用于获取方法的参数名称
        LocalVariableTableParameterNameDiscoverer u = new LocalVariableTableParameterNameDiscoverer();
        //获取方法的 全部的参数 名称
        String[] params = u.getParameterNames(clazz.getMethod(methodName, paramTypes));
        log4jutils.logger.info("参数名称:" + Arrays.toString(params));
        methodParName = Arrays.toString(params);

        //获方法的返回值类型
        Type t = clazz.getMethod(methodName, paramTypes).getAnnotatedReturnType().getType();
        log4jutils.logger.info("方法的返回值类型:" + t);
        methodReturn = t.toString();

        log4jutils.logger.info("执行前置通知了");

        //获取请求方法全限定名  包括参数 参数名称

    }

    //后置通知  如果出现异常就不执行
    @After("execution(* cn.xh.web.*.*(..))")
    public void doAfter(JoinPoint jp) throws Exception {
        long time = new Date().getTime() - visitTime.getTime(); //获取访问的时长
        String url = "";
        //获取url
        url = request.getScheme() + "://" + request.getServerName()
                + ":" + request.getServerPort()
                + request.getServletPath();
        if (request.getQueryString() != null) {
            url += "?" + request.getQueryString();
        }
        log4jutils.logger.info("url:" + url);
        log4jutils.logger.info("方法全路径" + method);
        //获取用户ip地址
        String ip = getRemoteHost(request);
        //获取当前操作的用户
        SecurityContext context = SecurityContextHolder.getContext();//从上下文中获了当前登录的用户
        User user = (User) context.getAuthentication().getPrincipal();
        String username = user.getUsername();

        //将日志相关信息封装到SysLog对象
        SysLog sysLog = new SysLog();
        sysLog.setExecutionTime(time); //执行时长
        sysLog.setIp(ip);
        sysLog.setMethod(method);
        sysLog.setUrl(url);
        sysLog.setUsername(username);
        sysLog.setVisitTime(visitTime);
        sysLog.setMethodParType(methodParType);
        sysLog.setMethodParName(methodParName);
        sysLog.setMethodParvalue(methodParvalue);
        sysLog.setMethodReturn(methodReturn);
        //调用Service完成插入操作
        sysLogService.save(sysLog);
        log4jutils.logger.info("执行最终通知");
    }


    /**
     * 获取远程客户端Ip
     *
     * @param request
     * @return
     */
    private String getRemoteHost(HttpServletRequest request) {
        String ip = request.getHeader("x-forwarded-for");
        if (ip == null || ip.length() == 0 || "unknown".equalsIgnoreCase(ip)) {
            ip = request.getHeader("Proxy-Client-IP");
        }
        if (ip == null || ip.length() == 0 || "unknown".equalsIgnoreCase(ip)) {
            ip = request.getHeader("WL-Proxy-Client-IP");
        }
        if (ip == null || ip.length() == 0 || "unknown".equalsIgnoreCase(ip)) {
            ip = request.getRemoteAddr();
        }
        return ip.equals("0:0:0:0:0:0:0:1") ? "127.0.0.1" : ip;
    }
}

Controller层

然后就是 Controller层

@Controller
@RequestMapping("/sysLog")
public class SysLogController {
        @Resource(name = "SysLogServiceImpl")
        private ISysLogService sysLogService;

        @RequestMapping("/findAll.do")
        public ModelAndView findAll() throws Exception {
            ModelAndView mv=new ModelAndView();
            List<SysLog> sysLogList= sysLogService.findAll();
            mv.addObject("sysLogs",sysLogList);
            mv.setViewName("list/syslog-list");
            return mv;
        }
}
