java-IOC容器

使用过Spring 的IOC容器 控制反转 那么这个就很好理解 原理差不多 就是把创建对象的过程交给容器来管理

需要用到的技术 反射, 单例模式 和 原型模式(浅拷贝和深度拷贝)

有两种配置文件反射方式

  1. 使用properties配置文件

  2. 使用xml配置文件

使用properties比较有局限性 因为就只有key 和value

而使用xml就非常灵活 自己根据情况而定 一般都是使用xml

案例结构:

下面案例需要的类和接口以及抽象类


public interface IAccountDao {
    void saveAccount();
}


public abstract class Absprototype implements Serializable,Cloneable {

    //浅度拷贝
    @Override
    public Object clone()  {
        Object object = null;
        try {
            object= super.clone();
        } catch (CloneNotSupportedException e) {
            e.printStackTrace();
        }
        return object;
    }

    //深度拷贝
    public Object  deepClone() {
        Object  object=null;
        try {
            // 序列化
            ByteArrayOutputStream bos = new ByteArrayOutputStream();
            ObjectOutputStream oos = new ObjectOutputStream(bos);
            oos.writeObject(this);
            // 反序列化
            ByteArrayInputStream bis = new ByteArrayInputStream(bos.toByteArray());
            ObjectInputStream ois = new ObjectInputStream(bis);
            object = ois.readObject();
        } catch (IOException e) {
            e.printStackTrace();
        } catch (ClassNotFoundException e) {
            e.printStackTrace();
        }
        return object;
    }
}

public class AccountDaoJDBCImpl extends Absprototype implements IAccountDao {

properties模式

properties配置文件 benas.properties

等号左边是类名 等号右边是类的包路径

AccountDaoJDBCImpl=cn.bdqn.impl.AccountDaoJDBCImpl
AccountDaoMybatisImpl=cn.bdqn.impl.AccountDaoMybatisImpl

将配置文件放入Resource目录里 在IDEA中还要设置此目录为资源文件夹否则找不到

容器类

public class Factory {

    //存储 对象 的容器
    private static Map<String,Object> map=new HashMap<String,Object>();
    static{
        Properties pro=new Properties();
        InputStream resource=Factory.class.getClassLoader().getResourceAsStream("benas.properties");
        try {
            pro.load(resource);

            Enumeration<Object> keys = pro.keys();
            while (keys.hasMoreElements()){
                String key=(String)keys.nextElement();
                String value=pro.getProperty(key);
                Object o = Class.forName(value).newInstance();
                map.put(key,o);
            }
        } catch (Exception e) {
            e.printStackTrace();
        }
    }

    /**
     * 一个参数 是 单例
     * @param name 需要的对象名称
     * @return  Object
     */
    public static Object getBean(String name)  {
        return map.get(name) ;
    }


    /**
     * 两个参数  单例 多例
     * @param name  需要的对象名称
     * @param pattern  pattern=true 多例(原型模式 默认深度拷贝)    pattern=false 单例
     * @return   Object
     */
    public static Object getBean(String name,Boolean pattern)  {
        if (pattern){
            Absprototype abs=(Absprototype) map.get(name);
            return  abs.deepClone() ;
        }else{
            return map.get(name) ;
        }

    }

    /**
     * 三个参数   单例 多例(深度拷贝或浅度拷贝)
     * @param name  需要的对象名称
     * @param pattern  pattern=true 多例(原型模式 默认深度拷贝)    pattern=false 单例
     * @param copy  true(浅度拷贝) false(深度度拷贝)
     * 建议: 如果对象内没有引用类型的成员变量(字符串包装类除外) 建议使用浅度拷贝 加快创建对象的速度
     * @return Object
     */
    public static Object getBean(String name,Boolean pattern,Boolean copy)  {
        if (pattern){

            if(copy){
                Absprototype abs=(Absprototype) map.get(name);
                return  abs.clone();
            } else {
                Absprototype abs = (Absprototype) map.get(name);
                return abs.deepClone();
            }
        } else {
            return map.get(name);
        }

    }

}

配置文件beans.xml

也是放入Resource资源目录里的

name=类名 (必须的)

class=”包路径” (必须的)

pattern=true(多例) 默认(深度拷贝) / 不写此属性默认为(单例)

prototype=”true”(浅度拷贝) 写此属性前必须拥有pattern=true 否则无效

如果你还想添加其他 属性的话 那么 需要修改 Factory工厂类里的代码 加上你的逻辑

<?xml version="1.0" encoding="UTF-8" ?>
<beans>
    <!--     默认单例 -->
    <bean name="AccountDaoJDBCImpl"     class="cn.bdqn.impl.AccountDaoJDBCImpl" />
    <!--pattern="true"多例(默认深度拷贝)`   -->
    <bean name="AccountDaoJDBCImpl1"     class="cn.bdqn.impl.AccountDaoJDBCImpl" pattern="true" />
    <!--pattern="true"多例     prototype=true(浅度拷贝)    必须pattern为true的时候才能使用浅度拷贝
        建议: 如果对象内没有引用类型的成员变量(字符串包装类除外) 建议使用浅度拷贝 加快创建对象的速度
     -->
    <bean name="AccountDaoJDBCImpl2"     class="cn.bdqn.impl.AccountDaoJDBCImpl" pattern="true"  prototype="true"/>

</beans>

容器类


import cn.bdqn.Absprototype;
import org.dom4j.*;
import org.dom4j.io.SAXReader;

import java.io.*;
import java.util.*;

import static cn.bdqn.factory.Factory.Moshi.*;

public class Factory {
    /**\
     *  SINGLE 单例
     *  MANGSALLOW 多例 浅度克隆
     *  MANGDEEPNESS 多例 深度克隆
     */
    enum Moshi{
        SINGLE, MANGSALLOW,MANGDEEPNESS
    }

  //存储对象的容器
    private static Map<String,Object> map=new HashMap<String,Object>();
    //xml 配置文件内 的配置信息
    private static Map<String,Map<String,String>> maps= new HashMap<>();
    //xml 配置文件内的配置信息组合  所对应的枚举模式
    private static Map<String,Moshi> moshi= new HashMap<>();
    static {
        SAXReader reader = new SAXReader();
//获取 Document对象
        Document document = null;
        InputStream resourceAsStream = Factory.class.getClassLoader().getResourceAsStream("beans.xml");
        try {
            document = reader.read(resourceAsStream);
        } catch (DocumentException e) {
            e.printStackTrace();
        }

        List list = document.selectNodes("/beans/bean");

        for (Iterator it = list.iterator(); it.hasNext();) {
            Element stuChild = (Element) it.next();//节点对象
            // 使用Element节点对象 可以获取所有此节点上的 属性名和值
            List<Attribute> attributes = stuChild.attributes();
            Map<String,String> map1=new HashMap<>();
            for (Attribute attribute : attributes) {
                map1.put(attribute.getName(),attribute.getValue());
            }
            maps.put(map1.get("name"),map1);
        }
        for (String stringStringMap : maps.keySet()) {
            try {
                map.put(stringStringMap,Class.forName(maps.get(stringStringMap).get("class")).newInstance());
            } catch (InstantiationException | IllegalAccessException | ClassNotFoundException e) {
                e.printStackTrace();
            }
        }


        Set<String> strings = maps.keySet();
        for (String string : strings) {
                Map<String, String> mapxml = maps.get(string);
                //判断是否存在pattern 节点属性
                String pattern =mapxml.get("pattern")==null? "false": "true";
                boolean patternPd=pattern.equals("true");
                //判断是否存在prototype 节点属性
                String prototype =mapxml.get("prototype")==null? "false": "true";
                boolean prototypePd = prototype.equals("true");

                if (patternPd&&prototypePd){ //开启多例  浅度克隆
                    moshi.put(string, MANGSALLOW);
                }else if(patternPd){//开启多例  默认深度拷贝
                    moshi.put(string, MANGDEEPNESS);
                }else{
                    moshi.put(string, SINGLE);
                }
        }
    }



    /**
     * @param name 需要的对象名称
     * @return  Object
     */
    public static Object getBean(String name)  {
        Set<String> strings = moshi.keySet();
        Object re=null;
        for (String string : strings) {
            if(string.equals(name)){
                switch (moshi.get(name)){
                    case SINGLE: //单例
                        re= map.get(name) ;
                        break;
                    case MANGSALLOW:  //多例 浅度拷贝
                        Absprototype abs=(Absprototype) map.get(name);
                        re= abs.clone() ;
                        break;
                    case MANGDEEPNESS:  //多例 深度拷贝
                        Absprototype abs1=(Absprototype) map.get(name);
                        re= abs1.deepClone() ;
                        break;
                    default:
                        re= map.get(name) ;
                        break;
                }
            }
        }
        return  re;

    }

}
