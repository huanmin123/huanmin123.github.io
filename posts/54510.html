Mybatis-PageHelper分页

PageHelper使用介绍

PageHelper是国内非常优秀的一款开源的mybatis分页插件，它支持基本主流与常用的数据库，例如mysql、
oracle、DB2、SQLite、Hsqldb等。

使用PageHelper需要添加插件 我们这里使用Maven添加

<!--分页插件-->
    <dependency>
      <groupId>com.github.pagehelper</groupId>
      <artifactId>pagehelper</artifactId>
      <version>5.1.2</version>
    </dependency>
    <dependency>
      <groupId>com.github.jsqlparser</groupId>
      <artifactId>jsqlparser</artifactId>
      <version>0.9.5</version>
    </dependency>

还需要将插件配置到 配置文件里才能使用

//如果使用的是ssm或者需要在<bean id="sqlSessionFactory" class="org.mybatis.spring.SqlSessionFactoryBean">里添加 

        <!-- 传入PageHelper的插件   分页 -->
        <property name="plugins">
            <array>
                <!-- 传入插件的对象 -->
                <bean class="com.github.pagehelper.PageInterceptor">
                    <property name="properties">
                        <props>
                            <prop key="helperDialect">mysql</prop>
                            <prop key="reasonable">true</prop>
                        </props>
                    </property>
                </bean>
            </array>
        </property>


如果只是使用Mybatis那么在核心配置文件中 添加

    <plugins>
        <!-- com.github.pagehelper为PageHelper类所在包名 -->
        <plugin interceptor="com.github.pagehelper.PageInterceptor">
            <!-- 使用MySQL的分页 -->
            <property name="helperDialect" value="mysql"/>
            <property name="pageSizeZero" value="true"/>
        </plugin>
 </plugins>

以上都完成后 就能使用分页了

  1. PageHelper.startPage(1,5);静态方法，传递两个参数（当前页码，每页查询条数）
  2. 使用pageHelper 分页的时候，不再关注分页语句，查询全部的语句
  3. 自动的对PageHelper.startPage 方法下的第一个sql 查询进行分页

一般将分页写在service层的 查询 sql 语句上

 @Override
    public List<Orders> findAll(int page,int size) throws Exception {
        PageHelper.startPage(page,size);  //进行分页
        return dao.findAll();
    }

然后表现成通过 PageInfo进行获取 分页的Bean信息

       List<Orders> all = ordersService.findAll(page,size); //数据库查询出来的数据

//通过  PageInfo 来获取 所有的分页信息
        PageInfo<Orders> pageInfo=new PageInfo<>(all);
//然后将pageInfo 发送给页面     通过 pageInfo里的list就能获取到分页的数据

service层(重点)

上面的看不懂没关系 主要就是这里 将查询出来的全部数据进行分页

public interface IOrdersService {

    public List<Orders> findAll(int page,int size) throws Exception;
}


import cn.xh.dao.IOrdersDao;
import cn.xh.eneity.Orders;
import cn.xh.service.IOrdersService;
import com.github.pagehelper.PageHelper;
import org.springframework.stereotype.Service;

import javax.annotation.Resource;
import java.util.List;

@Service("IOrdersServiceImpl")
public class IOrdersServiceImpl implements IOrdersService {
    @Resource(name = "IOrdersDao")
    private IOrdersDao dao;

    @Override
    public List<Orders> findAll(int page, int size) throws Exception {
        //在需要分页的sql语句前调用PageHelper进行分页
        PageHelper.startPage(page, size);
        return dao.findAll();
    }
}

web层

Controller控制器 这里将分页后的数据存入PageInfo里

PageInfo就是分页的各种信息上面有介绍


import cn.xh.eneity.Orders;
import cn.xh.service.IOrdersService;
import com.github.pagehelper.PageInfo;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestParam;
import org.springframework.web.servlet.ModelAndView;
import javax.annotation.Resource;
import java.util.List;

@Controller
@RequestMapping(path = "/orders")
public class OrdersController {
    @Resource(name = "IOrdersServiceImpl")
    private IOrdersService ordersService;

    @RequestMapping("/findAll.do.list")
    public ModelAndView findAll(
//            页码   如果没有值 默认 1
            @RequestParam(name = "page", required = true, defaultValue = "1") Integer page,
//            显示条数     如果没有值 默认 4
            @RequestParam(name = "size", required = true, defaultValue = "4") Integer size
    ) throws Exception {
        List<Orders> all = ordersService.findAll(page, size);
        //获取分页的全部信息
        PageInfo<Orders> pageInfo = new PageInfo<>(all);
        ModelAndView mod = new ModelAndView();
        mod.addObject("orderlist", pageInfo);
        mod.setViewName("orders-list");
        return mod;
    }

}
